第一步,添加库

1，cocopods安装三方库

1集成核心库（必须），在podfile文件中添加如下代码（如果没有podfile文件，则使用命令行工具进入到项目根目录执行 pod init ）

use_frameworks!
pod 'LTGameSDKCore'
pod 'AFNetworking'

2按需导入对应的第三方库

pod 'LTGameSDKCore/LTSDKPlatforms/Facebook'
pod 'LTGameSDKCore/LTSDKPlatforms/Google'
pod 'LTGameSDKCore/LTSDKPlatforms/Phone'
pod 'LTGameSDKCore/LTSDKPlatforms/QQ'
pod 'LTGameSDKCore/LTSDKPlatforms/Weixin'

3按需导入对应的UI库（具体地址请在线沟通）

pod 'SDKUIXXX'

2，info.plist添加如下字段(使应用可以访问http开头的网络请求)

<key>NSAppTransportSecurity</key>
    <dict>
        <key>NSAllowsArbitraryLoads</key>
        <true/>
    </dict>

第二步，初始化LT应用

1，在APPdelegate.h文件中导入头文件

#import <LTGameSDKCore/LTLoginManager.h>

2,初始化应用(可以在AppDelegate.m文件的didFinishLaunchingWithOptions方法中调用)
上线前请确认使用正式服务器接口地址

    [self.loginManager registLTAppID:@"从乐推平台申请的应用ID"
                              appkey:@"从乐推平台申请的应用AppKey"
                           serverURL:@"服务器地址"];

3,在AppDelegate.m文件添加应用回调

-(BOOL)application:(UIApplication *)app
           openURL:(NSURL *)url
           options:(NSDictionary<UIApplicationOpenURLOptionsKey,id> *)options{
    [[LTLoginManager sharedInstance] LTPlatformApplication:app openURL:url ptions:options];
    return YES;
}

登录状态检测

    [[LTLoginManager sharedInstance] getUserStateWithServerCheck:^(LTUser * _Nonnull loginUser) {
        if (loginUser.stateCode == LTLoginSuccess) {
            NSString *lt_UID = loginUser.LT_UID;

一 Facebook集成

1,添加URL scheme（URL scheme的添加方法见上文补充说明）
   值为fb123456789 (数字替换为从Facebook平台申请的APPID)

2,在info.plist中dict节点下添加

     <key>FacebookAppID</key>
     <string>此处填写从Facebook申请的APPID</string>
     <key>FacebookDisplayName</key>
     <string>应用的名称</string>
     <key>LSApplicationQueriesSchemes</key>
     <array>
     <string>fbapi</string>
     <string>fb-messenger-share-api</string>
     <string>fbauth2</string>
     <string>fbshareextension</string>
     </array>

3，配置-ObjC（配置方法见上文的补充说明）

4，在要使用登录功能的UIViewController中导入头文件

#import <LTGameSDKCore/LTLoginManager.h>
#import <LTGameSDKCore/LTUser.h>

5，在需要调用的地方调用登录

[[LTLoginManager sharedInstance] signInWithType:LTPlatformFacebook fromViewController:self callback:^(LTUser *loginUser) {
        if (loginUser.stateCode == LTLoginSuccess) {
            NSString *lt_UID = loginUser.LT_UID;
            NSString *LTUidToken = loginUser.LTUidToken;
            NSLog(@"LT_UID is %@ ::: LT_UID_Token is %@",lt_UID,LTUidToken);
        }else if (loginUser.stateCode == LTLoginCancel){
            NSLog(@"用户取消了登录");
        }else if (loginUser.stateCode == LTLoginFailed){
            NSLog(@"登录失败");
        }
    }];

第五步，手机登录相关

在要使用手机登录相关功能的.m文件中导入头文件

#import <LTPlatformPhone/LTPhoneManager.h>

1,发送验证码

    //获取输入的手机号
    NSString *phoneNumber = self.phoneInput.text;
    [[LTPhoneManager sharedInstance] sendPhoneCode:^(NSDictionary * _Nonnull resultInfo) {
        NSString *authCode = resultInfo[@"data"][@"auth_code"];
        NSString *msg = resultInfo[@"msg"];
    } phoneNumber:phoneNumber];

2,手机号注册

    //获取输入的手机号
    NSString *phoneNumber = self.phoneInput.text;
    //获取输入验证码
    NSString *authCode = self.authCodeInput.text;
    //获取输入密码
    NSString *passsword = self.passwordInput.text;
    [[LTPhoneManager sharedInstance] phoneRegist:^(LTUser * _Nonnull loginUser) {
        if (loginUser.stateCode == LTLoginSuccess) {
            NSString *lt_UID = loginUser.LT_UID;
            NSString *LTUidToken = loginUser.LTUidToken;
            NSLog(@"LT_UID is %@ ::: LT_UID_Token is %@",lt_UID,LTUidToken);
        }
    } withPhoneNumber:phoneNumber withAuthCode:authCode withPassword:passsword];

3,手机号登录

    //获取输入的手机号
    NSString *phoneNumber = self.phoneInput.text;
    //获取输入的密码
    NSString *passsword = self.passwordInput.text;
    [[LTPhoneManager sharedInstance] phoneLogin:^(LTUser * _Nonnull loginUser) {
        if (loginUser.stateCode == LTLoginSuccess) {
            NSString *lt_UID = loginUser.LT_UID;
            NSString *LTUidToken = loginUser.LTUidToken;
            NSLog(@"LT_UID is %@ ::: LT_UID_Token is %@",lt_UID,LTUidToken);
        }
    } withPhoneNumber:phoneNumber withPassword:passsword];

4,重置密码


    //获取输入的手机号
    NSString *phoneNumber = self.phoneInput.text;

    NSString *authCode = self.authCodeInput.text;

        NSString *msg = resultInfo[@"msg"];
